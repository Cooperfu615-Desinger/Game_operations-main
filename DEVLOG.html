關鍵模式：async 函式標準寫法

所有涉及 loading state 的 async 操作，須統一採用此結構，確保 Real API 下 flag 一定歸零：

const saveDraft = async () => {
  if (!dform.name.trim()) { message.warning('請填寫名稱'); return }
  saving.value = true
  try {
    const res = await api.save({ ...dform, created_by: authStore.user?.name ?? 'operator' })
    if (res.code === 0) { message.success('已儲存'); showDrawer.value = false; loadAll() }
  } catch {
    message.error('儲存失敗')
  } finally {
    saving.value = false   // ← 不論成功 / 失敗 / 例外，一定執行
  }
}
